Code used to create tiles

The tiles were created using shaders.

Tile B: Half circle

float sdCircle( vec2 uv, float r) { return length(uv) - r; } float s = sdCircle( uv-vec2(0.5, 0.0), 0.5 ); float m = S(0.008, 0.0, s);

Tile C: Half Square

//From Inigo Quilez float sdBox( vec2 uv, vec2 b ) { vec2 d = abs(uv)-b; return length(max(d,0.0)) + min(max(d.x,d.y),0.0); }
float s1 = sdBox( uv - vec2(0.5, 0.0), vec2(0.5, 0.5)); float m1 = S(0.008, 0.0, s1);
